  <script>
    // p5.js for starry background
    function setup() {
      let canvas = createCanvas(windowWidth, windowHeight);
      canvas.position(0, 0);
      canvas.style('z-index', '-1');
    }

    let stars = [];
    function draw() {
      background(0, 50);
      for (let star of stars) {
        fill(255, 255, 255, star.alpha);
        noStroke();
        ellipse(star.x, star.y, star.size);
        star.alpha += random(-5, 5);
        if (star.alpha < 0) star.alpha = 0;
        if (star.alpha > 255) star.alpha = 255;
      }
    }

    function windowResized() {
      resizeCanvas(windowWidth, windowHeight);
    }

    for (let i = 0; i < 100; i++) {
      stars.push({
        x: random(width),
        y: random(height),
        size: random(2, 5),
        alpha: random(100, 255)
      });
    }

    // Typing effect for message
    const message = "Dear Annu, today is all about you! Your smile lights up my world like the stars above. Here’s a shayari for you: \n\n‘Tere chehre ka noor, meri duniya ka rang, \nHar pal tere saath, jaise khushi ka sang.’ \n\nHappy Birthday, my love!";
    let i = 0;
    const messageDiv = document.getElementById('message');
    function typeMessage() {
      if (i < message.length) {
        messageDiv.innerHTML += message.charAt(i) === '\n' ? '<br>' : message.charAt(i);
        i++;
        setTimeout(typeMessage, 50);
      }
    }
    typeMessage();

    // Gallery animation
    const galleryBtn = document.getElementById('gallery-btn');
    const galleryImages = document.querySelectorAll('#gallery > div');
    galleryBtn.addEventListener('click', () => {
      galleryImages.forEach((img, index) => {
        setTimeout(() => {
          img.classList.add('visible');
        }, index * 500);
      });
      galleryBtn.classList.add('hidden');
    });

    // Confetti button
    document.getElementById('confetti-btn').addEventListener('click', () => {
      confetti({
        particleCount: 100,
        spread: 70,
        origin: { y: 0.6 }
      });
    });

    // Music toggle
    const audio = document.getElementById('bg-music');
    const musicToggle = document.getElementById('music-toggle');
    audio.play().catch(() => console.log("Autoplay prevented"));
    musicToggle.addEventListener('click', () => {
      if (audio.paused) {
        audio.play();
        musicToggle.textContent = "Pause Music";
      } else {
        audio.pause();
        musicToggle.textContent = "Play Music";
      }
    });
  </script>
